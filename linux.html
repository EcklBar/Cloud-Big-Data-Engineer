עבודה עם תיקיות

pwd

הצגת הנתיב המלא של התיקייה הנוכחית

pwd # מציג את הנתיב המלא של התיקייה הנוכחית
ls

הצגת תוכן התיקייה

ls # רשימה בסיסית
ls -l or ll # רשימה מפורטת עם פרטים
ls -la # כולל קבצים נסתרים
ls -lh # גדלים בפורמט קריא לאדם
ls -lt # ממוין לפי זמן שינוי
ls -lS # ממוין לפי גודל
ls -lR # רקורסיבי
cd

מעבר בין תיקיות

cd /path/to/dir # מעבר לתיקייה ספציפית
cd .. # מעבר לתיקייה האב
cd ~ # מעבר לתיקיית הבית
cd - # מעבר לתיקייה הקודמת
mkdir

יצירת תיקיות חדשות

mkdir new_folder # יצירת תיקייה חדשה
mkdir -p dir1/subdir1 # יוצר נתיב מלא
mkdir dir1 dir2 # יצירת מספר תיקיות
tree

הצגת מבנה תיקיות בצורת עץ

tree # הצגת מבנה התיקיות
tree -L 2 # עומק 2 רמות
📄

עבודה עם קבצים

touch

יצירת קבצים או עדכון זמן

touch file.txt # יצירת קובץ חדש
cp

העתקת קבצים ותיקיות

cp source dest # העתקת קובץ
cp -r folder1 folder2 # העתקת תיקייה
cp *.txt backup/ # העתקת כל קבצי הטקסט
mv

העברה ושינוי שם

mv oldname newname # שינוי שם
mv file /path/to/ # העברת קובץ
rm

מחיקת קבצים ותיקיות

rm file.txt # מחיקת קובץ
rm -i file.txt # מחיקה עם אישור
rm -rf folder/ # מחיקה כפויה לתיקייה (זהירות!)
find

חיפוש קבצים מתקדם

find . -name "*.py" # חיפוש קבצי Python
find / -size +100M # קבצים גדולים מ-100MB
find . -mtime -7 # שונו בשבוע האחרון
📝

עבודה עם תוכן קבצים

cat / tac

הצגת תוכן מלא

cat file.txt # הצגת כל התוכן
cat f1 f2 > merged # איחוד קבצים
tac file.txt # הצגה מהסוף להתחלה
head / tail

הצגת תחילת/סוף קובץ

head -n 5 file.csv # 5 שורות ראשונות (לבדיקת כותרות)
tail -f log.txt # מעקב חי אחר לוגים
shuf

דגימה רנדומלית (שימושי ל-Data)

shuf -n 5 data.csv # 5 שורות אקראיות מקובץ
grep / rg

חיפוש תבניות בטקסט

grep "error" log.txt # חיפוש שגיאות
grep -r "config" . # חיפוש בתיקייה
rg "pattern" # חיפוש מהיר עם ripgrep
wc

ספירת מילים ושורות

wc -l file.csv # ספירת רשומות בקובץ
sort / uniq

מיון וסינון כפילויות

sort file.txt | uniq # הסרת כפילויות
sort -n # מיון מספרי
🔤

עבודה עם משתנים

הגדרת משתנים

יצירה והגדרת משתנים

VAR="value" # משתנה מקומי
export VAR="value" # משתנה סביבה
echo $VAR # הצגת ערך
משתני מערכת

משתנים חשובים

$HOME # תיקיית הבית
$USER # שם המשתמש
$PATH # נתיבי חיפוש לתוכניות
🔄

הפניית קלט/פלט

הפניות > >>

ניתוב פלט לקבצים

cmd > file.txt # כתיבה לקובץ (דורס)
cmd >> file.txt # הוספה לקובץ
cmd 2> error.log # הפניית שגיאות (stderr)
🔗

צינורות ומסננים

pipe |

חיבור בין פקודות

cat data.csv | wc -l # ספירת שורות בפלט
ps aux | grep python # סינון תהליכים
awk

עיבוד טקסט מתקדם (חובה ל-DE)

awk -F',' '{print $2}' file.csv # הדפסת עמודה 2 מקובץ CSV
awk '{sum+=$3} END {print sum}' # סכום עמודה
sed

עריכת זרם (החלפה בטקסט)

sed 's/old/new/g' file # החלפת טקסט
sed -i 's/ERROR/WARNING/g' file # החלפה בתוך הקובץ (זהירות!)
✏️

עורך VI/VIM

פקודות בסיס

עריכה ושמירה

i # מצב עריכה (Insert)
Esc # יציאה למצב פקודה
:wq # שמירה ויציאה
:q! # יציאה ללא שמירה
dd # מחיקת שורה
🔐

הרשאות קבצים

chmod

שינוי הרשאות

chmod +x script.sh # הוספת הרצת
chmod 755 file # rwx-rx-rx
chmod 600 key.pem # הרשאה למפתחות פרטיים (רק בעלים)
chmod 400 key.pem # קריאה בלבד לבעלים
chown

שינוי בעלות

chown user:group file # שינוי בעלים וקבוצה
chown -R user dir/ # שינוי רקורסיבי
💻

מידע מערכת

disk usage

מקום בדיסק וזיכרון

df -h # שטח דיסק פנוי
du -sh folder/ # גודל של תיקייה ספציפית
du -h --max-depth=1 # גודל תתי-תיקיות (חשוב לניקוי)
free -h # מצב זיכרון RAM
top / htop

ניטור תהליכים

top # ניטור בזמן אמת
htop # גרסה משופרת וצבעונית
ps aux | grep python # מציאת תהליך ספציפי
kill -9 PID # הרג תהליך תקוע

קיצורי מקלדת

קיצור פעולה תיאור

Ctrl + E סוף שורה מעבר לסוף השורה הנוכחית

Ctrl + R חיפוש בהיסטוריה חיפוש בפקודות קודמות
Tab השלמה אוטומטית השלמת שמות קבצים ופקודות

טיפים למהנדסי נתונים

פקודות חיוניות למהנדסי נתונים

# בדיקת כותרות קובץ CSV ענק
head -n 5 large_file.csv
# דגימת שורות אקראיות (לבדיקת איכות נתונים)
shuf -n 10 data.csv
# מציאת התיקייה שתופסת את המקום בדיסק (Hadoop/Logs)
du -h --max-depth=1 /var/log | sort -hr
# עריכת קובץ קונפיגורציה בזהירות
cp config.xml config.xml.bak && vi config.xml
