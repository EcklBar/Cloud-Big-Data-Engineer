🐧 Linux למהנדסי נתונים

המדריך האינטראקטיבי המלא

📁 עבודה עם תיקיות

הבסיס של מערכת הקבצים בלינוקס. תיקיות מסודרות במבנה עץ המתחיל ב-/.

pwd

הצגת הנתיב הנוכחי (איפה אני נמצא?)

pwd
ls

הצגת תוכן התיקייה

ls -la # רשימה מפורטת כולל נסתרים
ls -lh # גדלים בפורמט קריא
ls -lt # מיון לפי זמן שינוי
cd

מעבר בין תיקיות

cd /var/log # מעבר לנתיב מוחלט
cd .. # עליה רמה אחת למעלה
cd ~ # חזרה לתיקיית הבית
mkdir

יצירת תיקיות

mkdir new_folder
mkdir -p data/2023/jan # יצירת נתיב מלא

📄 עבודה עם קבצים

יצירה, העתקה, העברה ומחיקה של קבצים.

touch

יצירת קובץ ריק או עדכון זמן גישה

touch file.txt
cp

העתקת קבצים

cp source.txt dest.txt
cp -r data_dir backup_dir # העתקת תיקייה
mv

העברה או שינוי שם

mv old_name.txt new_name.txt
mv file.txt /tmp/
rm

מחיקת קבצים (בזהירות!)

rm file.txt
rm -rf temp_folder # מחיקה כפויה של תיקייה
find

חיפוש קבצים מתקדם

find . -name "*.csv" # כל קבצי ה-CSV
find /var/log -mtime -7 # שונו ב-7 ימים אחרונים
find . -size +100M # קבצים מעל 100 מגה

📝 תוכן קבצים

צפייה וניתוח תוכן של קבצים טקסטואליים.

cat / tac

הדפסת תוכן הקובץ

cat config.json
tac log.txt # הדפסה מהסוף להתחלה
head / tail

תצוגה חלקית

head -n 5 data.csv # הצגת כותרות (5 שורות ראשונות)
tail -f app.log # מעקב חי אחרי לוגים
grep

חיפוש טקסט בתוך קבצים

grep "ERROR" server.log
grep -r "db_host" . # חיפוש בכל התיקייה
wc

ספירת מילים ושורות

wc -l data.csv # כמה רשומות יש בקובץ?

🔄 הפניות וצינורות (Pipes)

שרשור פקודות הוא הכוח האמיתי של לינוקס בעיבוד נתונים.

Redirection >

שמירת פלט לקובץ

echo "Log Start" > log.txt # דריסת הקובץ
echo "New Entry" >> log.txt # הוספה לסוף הקובץ
Pipe |

העברת פלט מפקודה לפקודה

cat data.csv | wc -l
ps aux | grep python
history | grep "aws"
awk

עיבוד טקסט ושליפת עמודות (חובה ל-DE)

awk -F',' '{print $2}' file.csv # הדפסת עמודה שניה
ls -l | awk '{print $5, $9}' # גודל ושם קובץ
sort & uniq

מיון וסינון כפילויות

sort names.txt | uniq
sort -nr numbers.txt # מיון מספרי יורד

🔐 הרשאות (Permissions)

ניהול גישה לקבצים: קריאה (r), כתיבה (w), והרצה (x).

chmod

שינוי הרשאות

chmod +x script.sh # הפיכת סקריפט לניתן להרצה
chmod 755 file # בעלים: הכל, אחרים: קריאה/הרצה
chmod 400 key.pem # הגנה על מפתח פרטי
chown

שינוי בעלות

chown user:group file.txt
chown -R user /var/www # שינוי רקורסיבי

💻 ניטור מערכת

בדיקת משאבים, זיכרון ותהליכים רצים.

Disk & Memory

מקום בדיסק וזיכרון RAM

df -h # מקום פנוי בדיסק
du -sh folder/ # גודל של תיקייה ספציפית
free -h # מצב הזיכרון
top / htop

מנהל המשימות

top # צפייה בתהליכים בזמן אמת
htop # גרסה צבעונית ונוחה יותר
kill

סגירת תהליכים

kill 1234 # סגירת תהליך לפי ID
kill -9 1234 # סגירה כפויה (Force Kill)

💡 טיפים למהנדסי נתונים

פקודות קטנות שעושות הבדל גדול בעבודה עם Data.

בדיקת איכות נתונים מהירה

במקום לפתוח קובץ ענק עם Excel ולהתקע, השתמשו בטרמינל:

head -n 5 large.csv # בדיקת שמות העמודות
shuf -n 5 large.csv # דגימה אקראית של 5 שורות (Random Sample)
wc -l large.csv # ספירת סך כל הרשומות

SSH וחיבור מרחוק

התחברות לשרתים בצורה מאובטחת:

ssh user@server_ip
scp file.txt user@server_ip:/tmp # העתקת קובץ לשרת
