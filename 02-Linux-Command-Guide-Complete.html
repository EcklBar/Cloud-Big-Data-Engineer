עבודה עם תוכן קבצים

cat

הצגת תוכן מלא

cat file.txt # הצגת כל התוכן
cat file1.txt file2.txt # הצגת מספר קבצים
cat -n file.txt # עם מספור שורות
cat -b file.txt # מספור רק שורות לא ריקות
cat file1 file2 > merged # חיבור של כמה קבצים לקובץ אחד
cat >> file.txt # הוספה לקובץ (Ctrl+D לסיום)
cat new.txt >> full.txt # הוספת תוכן קובץ לקובץ אחר
tac

הצגת תוכן בסדר הפוך (מלמטה למעלה)

tac file.txt # הצגת קובץ מלמטה למעלה (cat הפוך)
echo -e "א\nב\nג" | tac # שימוש עם צינורות
less / more

צפייה מדורגת בקובץ

less file.txt # ניווט חופשי (מומלץ)
more file.txt # ניווט קדימה בלבד
# קיצורי less:
# Space/f=עמוד קדימה, b=אחורה
# g=תחילה, G=סוף, q=יציאה
# /pattern=חיפוש קדימה
# ?pattern=חיפוש אחורה
# n=הבא, N=קודם
head / tail

הצגת תחילת/סוף קובץ

head file.txt # 10 שורות ראשונות
head -n 20 file.txt # 20 שורות ראשונות
head -c 100 file.txt # 100 תווים ראשונים
tail file.txt # 10 שורות אחרונות
tail -n 50 file.txt # 50 שורות אחרונות
tail -f logfile.txt # מעקב חי אחר קובץ לוג
tail -F logfile.txt # מעקב עם טיפול בהחלפת קבצים
grep

חיפוש תבניות בטקסט

grep "pattern" file # חיפוש בסיסי
grep -i "pattern" file # חיפוש לא רגיש לאותיות
grep -n "pattern" file # עם מספרי שורות
grep -r "pattern" dir/ # חיפוש רקורסיבי
grep -v "pattern" file # שורות שלא מכילות תבנית
grep -c "pattern" file # ספירת התאמות
grep -E "regex" file # ביטויים רגולריים מורחבים
ripgrep (rg)

חיפוש מהיר ומתקדם - חלופה משופרת ל-grep

rg "search_term" # חיפוש בסיסי
rg -n "search_term" # חיפוש עם מספרי שורות
rg -i "PATTERN" # חיפוש case-insensitive
rg -C 3 "pattern" # הצגת הקשר (3 שורות מסביב)
rg -t py "import" # חיפוש רק בקבצי Python
rg -uuu "pattern" # ביטול כל הסינון האוטומטי
rg --hidden "pattern" # הכללת קבצים נסתרים
rg --binary "pattern" # חיפוש בקבצים בינאריים
rg -c "pattern" # ספירת התאמות
rg -i --color=always 'email.*com' # עם צביעה מודגשת
rg -i -l 'email.*com' # הצגת שמות קבצים בלבד
rg -i -o 'email.*com' # הצגת ההתאמות בלבד
rg -i -o 'email.*com' > found_emails.txt # שמירת התאמות
rg -i -o 'email.*com' | wc -l # ספירת התאמות
rg -i -o 'email.*com' | sort | uniq # הסרת כפילויות
wc

ספירת מילים ושורות

wc file.txt # שורות, מילים, תווים
wc -l file.txt # רק שורות
wc -w file.txt # רק מילים
wc -c file.txt # רק תווים
sort

מיון שורות

sort file.txt # מיון אלפביתי
sort -n file.txt # מיון נומרי
sort -r file.txt # מיון הפוך
sort -k 2 file.txt # מיון לפי עמודה 2
sort -u file.txt # מיון והסרת כפילויות
uniq

הסרת כפילויות

uniq file.txt # הסרת כפילויות רצופות
uniq -c file.txt # עם ספירה
uniq -d file.txt # רק שורות כפולות
uniq -u file.txt # רק שורות ייחודיות
sort file.txt | uniq -c | sort -nr # ספירת מילים וסידור לפי תדירות

השוואת קבצים

הפניית קלט/פלט

הפניית פלט

cmd > file.txt # כתיבה לקובץ (החלפה)
cmd >> file.txt # הוספה לקובץ

cmd > output.txt 2>&1 # הפניית שגיאות לאותו יעד של הפלט הרגיל
cmd &> all.log # הפניית פלט ושגיאות יחד (קיצור)
cmd > /dev/null # התעלמות מפלט ("פח האשפה")
cmd 2> /dev/null # התעלמות משגיאות בלבד

ps # תהליכים של המשתמש הנוכחי
ps aux # כל התהליכים מפורט
ps -ef # כל התהליכים (System V)
ps -u user # תהליכים של משתמש מסוים
uname

מידע מערכת

uname # שם המערכת
uname -a # כל המידע
uname -r # גרסת קרנל
hostname # שם המחשב
kill

סיום תהליכים

kill PID # סיום תהליך רגיל
kill -9 PID # סיום כפוי
killall process_name # סיום לפי שם
systemctl

ניהול שירותים

systemctl status service # סטטוס שירות
systemctl start service # הפעלת שירות
systemctl stop service # עצירת שירות
systemctl restart service # הפעלה מחדש
systemctl enable service # הפעלה אוטומטית

whoami / id

מידע משתמש נוכחי

whoami # שם המשתמש הנוכחי
id # מידע מלא על המשתמש
id username # מידע על משתמש אחר
su / sudo

החלפת משתמש והרשאות

su - username # החלפת משתמש
sudo command # הרצה עם הרשאות root
sudo -i # כניסה כ-root
sudo -u user command # הרצה כמשתמש אחר
passwd

שינוי סיסמה

passwd # שינוי סיסמה שלך
passwd username # שינוי סיסמה למשתמש (כ-root)
passwd -l user # נעילת משתמש
useradd / adduser

הוספת משתמש

useradd username # הוספת משתמש (בסיסי)
adduser username # הוספת משתמש (אינטראקטיבי)
useradd -m -s /bin/bash user # עם תיקיית בית ושל
groups

ניהול קבוצות

groups # הקבוצות שלך
groups username # קבוצות של משתמש
usermod -aG group user # הוספת משתמש לקבוצה

crontab

משימות מתוזמנות

crontab -e # עריכת משימות
crontab -l # רשימת משימות
crontab -r # מחיקת כל המשימות
0 2 * * * /script.sh # כל יום ב-2:00
*/5 * * * * command # כל 5 דקות
at

משימה חד פעמית

at 15:00 # תזמון ל-15:00
at now + 2 hours # בעוד שעתיים
atq # רשימת משימות ממתינות
atrm # ביטול משימה
watch

הרצה חוזרת

watch df -h # רענון כל 2 שניות
watch -n 5 command # רענון כל 5 שניות
watch -d ls -l # הדגשת שינויים
sleep

השהיית הרצה

sleep 10 # השהייה של 10 שניות
sleep 2m # השהייה של 2 דקות
sleep 1h # השהייה של שעה

קיצורי מקלדת

קיצור פעולה תיאור
Ctrl + A תחילת שורה מעבר לתחילת השורה הנוכחית
Ctrl + E סוף שורה מעבר לסוף השורה הנוכחית
Ctrl + C ביטול פקודה עצירת הפקודה הרצה
Ctrl + D יציאה סגירת הטרמינל או יציאה
Ctrl + L ניקוי מסך ניקוי הטרמינל (כמו clear)
Ctrl + R חיפוש בהיסטוריה חיפוש בפקודות קודמות
Ctrl + Z השהיית תהליך העברת התהליך לרקע
Ctrl + K מחיקה מהסמן לסוף מחיקת הטקסט מהסמן עד סוף השורה
Ctrl + U מחיקה מהסמן לתחילה מחיקת הטקסט מתחילת השורה עד הסמן
Ctrl + W מחיקת מילה מחיקת המילה האחרונה
Tab השלמה אוטומטית השלמת שמות קבצים ופקודות
↑ / ↓ היסטוריה ניווט בפקודות קודמות

טיפים למהנדסי נתונים

פקודות חיוניות למהנדסי נתונים

# עיבוד נתונים
grep, awk, sed, cut, sort, uniq, wc
# ניטור מערכת
top, htop, df, du, ps, kill
# רשת והעברת קבצים
wget, curl, scp, rsync
# דחיסה וארכיון
tar, gzip, zip, unzip
# תהליכים ברקע
nohup, screen, tmux, jobs, bg, fg

עקרונות חשובים

  • אוטומציה: השתמש בסקריפטים ו-pipes ליצירת תהליכי עיבוד נתונים אוטומטיים
  • תיעוד: השתמש בהערות ובקבצי לוג לתיעוד התהליכים
  • אבטחה: הגדר הרשאות מתאימות לקבצי נתונים רגישים
  • ארגון: צור מבנה תיקיות לוגי וקבוע לפרויקטים
  • גיבוי: השתמש בגרסאות וגיבויים של נתונים חשובים

טיפים מתקדמים

  1. השתמש ב-history לחזרה על פקודות קודמות
  2. למד shortcuts של מקלדת (Ctrl+A, Ctrl+E, Ctrl+R)
  3. צור aliases לפקודות נפוצות ב-~/.bashrc
  4. השתמש ב-tab completion לחיסכון בזמן
  5. למד regex לחיפושים מתקדמים עם grep ו-sed

פקודות נוספות שימושיות

# Command Substitution - החלפת פקודה
x=$(cat /tmp/file)
echo $x
# שרשור מחרוזות עם סוגריים מסולסלים
prefix="hello"
echo "${prefix}-world" # פלט: hello-world
# /dev/null = "פח האשפה" להתעלמות מפלט
command > /dev/null 2>&1
# tac - הצגת תוכן קובץ מלמטה למעלה (cat הפוך)
tac file.txt
echo -e "א\nב\nג" | tac
